30 void Gate::Validate() const {
31  // Detect inhibit flavor.
32  if (formula_->type() != kAnd || !Element::HasAttribute("flavor") ||
33  Element::GetAttribute("flavor").value != "inhibit") {
34  return;
35  }
36  if (formula_->num_args() != 2) {
37  throw ValidationError(Element::name() +
38  "INHIBIT gate must have only 2 children");
39  }
40  int num_conditional = boost::count_if(
41  formula_->basic_event_args(),
42  [](const BasicEventPtr& event) {
43  return event->HasAttribute("flavor") &&
44  event->GetAttribute("flavor").value == "conditional";
45  });
46  if (num_conditional != 1)
47  throw ValidationError(Element::name() + " : INHIBIT gate must have" +
48  " exactly one conditional event.");
49 }

59 void Formula::vote_number(int number) {
60  if (type_ != kVote) {
61  throw LogicError(
62  "The vote number can only be defined for 'atleast' formulas. "
63  "The operator of this formula is '" +
64  std::string(kOperatorToString[type_]) + "'.");
65  }
66  if (number < 2)
67  throw InvalidArgument("Vote number cannot be less than 2.");
68  if (vote_number_)
69  throw LogicError("Trying to re-assign a vote number");
70 
71  vote_number_ = number;
72 }

74 void Formula::Validate() const {
75  switch (type_) {
76  case kAnd:
77  case kOr:
78  case kNand:
79  case kNor:
80  if (num_args() < 2)
81  throw ValidationError("\"" + std::string(kOperatorToString[type_]) +
82  "\" formula must have 2 or more arguments.");
83  break;
84  case kNot:
85  case kNull:
86  if (num_args() != 1)
87  throw ValidationError("\"" + std::string(kOperatorToString[type_]) +
88  "\" formula must have only one argument.");
89  break;
90  case kXor:
91  if (num_args() != 2)
92  throw ValidationError("\"xor\" formula must have exactly 2 arguments.");
93  break;
94  case kVote:
95  if (num_args() <= vote_number_)
96  throw ValidationError("\"atleast\" formula must have more arguments "
97  "than its vote number " +
98  std::to_string(vote_number_) + ".");
99  }
100 }
