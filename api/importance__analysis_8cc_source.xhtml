36 void ImportanceAnalysis::Analyze() noexcept {
37  CLOCK(imp_time);
38  LOG(DEBUG3) << "Calculating importance factors...";
39  double p_total = this->p_total();
40  const std::vector<const mef::BasicEvent*>& basic_events =
41  this->basic_events();
42  std::vector<int> occurrences = this->occurrences();
43  for (int i = 0; i < basic_events.size() && occurrences[i]; ++i) {
44  const mef::BasicEvent& event = *basic_events[i];
45  double p_var = event.p();
46  ImportanceFactors imp;
47  imp.occurrence = occurrences[i];
48  imp.mif = this->CalculateMif(i);
49  imp.cif = p_var * imp.mif / p_total;
50  imp.raw = 1 + (1 - p_var) * imp.mif / p_total;
51  imp.dif = p_var * imp.raw;
52  imp.rrw = p_total / (p_total - p_var * imp.mif);
53  importance_.push_back({event, imp});
54  }
55  LOG(DEBUG3) << "Calculated importance factors in " << DUR(imp_time);
56  Analysis::AddAnalysisTime(DUR(imp_time));
57 }

81 double ImportanceAnalyzer<Bdd>::CalculateMif(const Bdd::VertexPtr& vertex,
82  int order, bool mark) noexcept {
83  if (vertex->terminal())
84  return 0;
85  Ite& ite = Ite::Ref(vertex);
86  if (ite.mark() == mark)
87  return ite.factor();
88  ite.mark(mark);
89  if (ite.order() > order) {
90  if (!ite.module()) {
91  ite.factor(0);
92  } else { /// @todo Detect if the variable is in the module.
93  // The assumption is
94  // that the order of a module is always larger
95  // than the order of its variables.
96  double high = RetrieveProbability(ite.high());
97  double low = RetrieveProbability(ite.low());
98  if (ite.complement_edge())
99  low = 1 - low;
100  const Bdd::Function& res =
101  bdd_graph_->modules().find(ite.index())->second;
102  double mif = CalculateMif(res.vertex, order, mark);
103  if (res.complement)
104  mif = -mif;
105  ite.factor((high - low) * mif);
106  }
107  } else if (ite.order() == order) {
108  assert(!ite.module() && "A variable can't be a module.");
109  double high = RetrieveProbability(ite.high());
110  double low = RetrieveProbability(ite.low());
111  if (ite.complement_edge())
112  low = 1 - low;
113  ite.factor(high - low);
114  } else {
115  assert(ite.order() < order);
116  double p_var = 0;
117  if (ite.module()) {
118  const Bdd::Function& res =
119  bdd_graph_->modules().find(ite.index())->second;
120  p_var = RetrieveProbability(res.vertex);
121  if (res.complement)
122  p_var = 1 - p_var;
123  } else {
124  p_var = prob_analyzer()->p_vars()[ite.index()];
125  }
126  double high = CalculateMif(ite.high(), order, mark);
127  double low = CalculateMif(ite.low(), order, mark);
128  if (ite.complement_edge())
129  low = -low;
130  ite.factor(p_var * high + (1 - p_var) * low);
131  }
132  return ite.factor();
133 }
