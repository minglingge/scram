33 /// Finds the installation directory on Windows.
34 ///
35 /// @returns POSIX path to the installation directory.
36 ///
37 /// @pre The main executable is located at "install_dir/bin/prog.exe".
38 ///
39 /// @note Exits the program with 1
40 /// if system calls fail.
41 ///
42 /// @warning The code must be compiled with Unicode support.
43 static std::string GetInstallPath() {
44  // The following code doesn't conform to TCHAR, DWORD Windows nonsense.
45  wchar_t buffer[MAX_PATH] = {};
46  int ret = GetModuleFileNameW(nullptr, buffer, MAX_PATH); // windows.h
47  if (ret == 0 || ret == MAX_PATH) {
48  std::cerr << "Failure to retrieve installation directory.\n"
49  << "OS error code: " << GetLastError() << std::endl;
50  std::exit(1);
51  }
52  return boost::filesystem::path(buffer) // executable
53  .parent_path() // bin
54  .parent_path() // install
55  .generic_string(); // POSIX format.
56 }
57 #else
58 /// @returns Path to the installation directory known at compile-time.

60 #endif
